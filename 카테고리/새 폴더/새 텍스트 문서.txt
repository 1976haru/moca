이 부분을 전부 제거하고 아래로 교체해줘.

  <button id="btn-gemini" onclick="selectAI('gemini')"

<script>
// AI 선택 및 localStorage 연동
var _selectedAI = localStorage.getItem('uc_ai_provider') || 'claude';

function selectAI(provider) {
  _selectedAI = provider;
  localStorage.setItem('uc_ai_provider', provider);
  updateAIButtons();
}

function updateAIButtons() {
  var providers = ['claude', 'openai', 'gemini'];
  var names = {claude:'🟣 Claude', openai:'🟢 OpenAI', gemini:'🔵 Gemini'};
  var keys = {
    claude: localStorage.getItem('uc_claude_key'),
    openai: localStorage.getItem('uc_openai_key'),
    gemini: localStorage.getItem('uc_gemini_key')
  };

  providers.forEach(function(p) {
    var btn = document.getElementById('btn-' + p);
    if (!btn) return;
    var hasKey = keys[p] && keys[p].trim().length > 0;
    var isSelected = _selectedAI === p;

    if (isSelected) {
      btn.style.background = 'linear-gradient(135deg,#9181ff,#b14d82)';
      btn.style.color = '#fff';
      btn.style.border = 'none';
      btn.textContent = names[p] + ' ✓';
    } else {
      btn.style.background = '#fff';
      btn.style.color = hasKey ? '#333' : '#bbb';
      btn.style.border = '1px solid ' + (hasKey ? '#ddd' : '#eee');
      btn.textContent = names[p] + (hasKey ? '' : ' 🔴');
    }
  });

  var status = document.getElementById('ai-status');
  if (status) {
    var key = keys[_selectedAI];
    status.textContent = (key && key.trim().length > 0)
      ? '🟢 연결됨'
      : '🔴 키 없음 — 설정에서 입력해주세요';
    status.style.color = (key && key.trim().length > 0) ? '#2f7a54' : '#e53';
  }
}

// API 키 가져오기 함수 (기존 getApiKey 대체)
function getApiKey(provider) {
  var p = provider || _selectedAI;
  var keyMap = {
    claude: 'uc_claude_key',
    openai: 'uc_openai_key',
    gemini: 'uc_gemini_key'
  };
  return localStorage.getItem(keyMap[p] || 'uc_claude_key') || '';
}

// 페이지 로드 시 상태 복원
document.addEventListener('DOMContentLoaded', function() {
  updateAIButtons();
});
updateAIButtons();
</script>

중요:
1. 기존에 API 키를 입력받아 저장하던 코드도 찾아서
   getApiKey() 함수를 위의 새 함수로 교체해줘.
2. 기존에 localStorage.getItem('api_key_v10') 또는
   localStorage.getItem('claude_api_key') 같은 코드를
   localStorage.getItem('uc_claude_key') 로 전부 교체해줘.
3. 대본 생성 시 API 키 없으면:
   alert 대신 토스트 메시지로:
   "⚠️ API 키가 없어요. 설정에서 입력해주세요"
   + 설정으로 이동 버튼 표시
