/* 상단바 상태 업데이트 */
function scriptUpdateTopbar(){
  var el = document.getElementById('script-topbar-status');
  if(!el) return;
  var genre = document.querySelector('.tnav.on')?.textContent?.trim() || '';
  var ch = document.querySelector('.ch-preset.on, .preset-btn.on')?.textContent?.trim() || '';
  var parts = [genre, ch].filter(Boolean);
  el.textContent = parts.length ? parts.join(' · ') : '대본생성기';
}

/* TAB 함수 원본 호출 후 상태바 업데이트 */
var _origTAB = window.TAB;
window.TAB = function(name, btn){
  if(typeof _origTAB === 'function') _origTAB(name, btn);
  scriptUpdateTopbar();
};

function scriptToggleAdvanced(){

  var open = el.style.display === 'none';

/* AI 패널 토글 */
function scriptToggleAiPanel(){
  var el = document.getElementById('script-ai-panel');
  var pp = document.getElementById('script-preset-panel');
  if(!el) return;
  if(pp) pp.style.display = 'none';
  el.style.display = el.style.display === 'none' ? 'block' : 'none';

  /* AI 버튼 렌더 */
  var wrap = document.getElementById('script-ai-btns');
  var label = document.getElementById('script-ai-selected-label');
  if(!wrap) return;
  var ais = [
    {id:'claude',  name:'Claude',  color:'#7C5CFC'},
    {id:'openai',  name:'OpenAI',  color:'#10A37F'},
    {id:'gemini',  name:'Gemini',  color:'#4285F4'}
  ];
  var current = localStorage.getItem('selected_ai') || 'claude';
  wrap.innerHTML = ais.map(function(a){
    var on = current === a.id;
    return '<button onclick="scriptSelectAi(\''+a.id+'\')" style="'+
      'border:2px solid '+(on?a.color:'var(--line)')+';'+
      'background:'+(on?a.color+'18':'#fff')+';'+
      'border-radius:999px;padding:7px 16px;'+
      'font-size:13px;font-weight:800;'+
      'color:'+(on?a.color:'var(--sub)')+';'+
      'cursor:pointer;transition:.15s;">'+
      a.name+'</button>';
  }).join('');
  if(label) label.textContent = '현재: ' + (ais.find(function(a){return a.id===current;})?.name||'Claude');
}

function scriptSelectAi(id){
  localStorage.setItem('selected_ai', id);
  /* 기존 AI 선택 버튼 연동 */

            document.querySelector('[onclick*="setAI(\''+id+'\'"]') ||
            document.querySelector('.ai-btn[data-ai="'+id+'"]');
  if(btn) btn.click();
  scriptToggleAiPanel();
}

/* from=shorts 파라미터 처리 */
(function(){
  var params = new URLSearchParams(location.search);
  if(params.get('from') !== 'shorts') return;

  /* 숏츠 완료 버튼 표시 */
  var doneBtn = document.getElementById('sc-shorts-done');
  if(doneBtn) doneBtn.style.display = 'block';

  /* 주제 자동 입력 */
  var topic = params.get('topic') || localStorage.getItem('sh_topic_bridge') || '';
  if(topic){
    window.addEventListener('DOMContentLoaded', function(){
      var inp = document.getElementById('one-input') ||
                document.querySelector('input[placeholder*="주제"]') ||
                document.querySelector('textarea[placeholder*="주제"]');
      if(inp && !inp.value) inp.value = topic;
    });
  }

  /* 탭 자동 선택 */
  var tab = params.get('tab');
  if(tab){
    window.addEventListener('DOMContentLoaded', function(){
      var btn = document.querySelector('.tnav[onclick*="\''+tab+'\'"]');
      if(btn) btn.click();
    });
  }

  /* API 키 브리지 */
  try {
    var ck = localStorage.getItem('uc_claude_key');
    var ok = localStorage.getItem('uc_openai_key');
    var gk = localStorage.getItem('uc_gemini_key');
    if(ck) localStorage.setItem('api_key_v10', ck);
    if(ok) localStorage.setItem('v30_openai_key', ok);
    if(gk) localStorage.setItem('v30_gemini_key', gk);
  } catch(e){}
})();

/* 결과창 자동 표시 */
(function(){
  var origOut = document.getElementById('out');
  if(!origOut) return;
  var observer = new MutationObserver(function(){
    var wrap = document.getElementById('sc-result-wrap');
    if(wrap && origOut.value.trim()){
      wrap.style.display = 'block';
      /* 글자수·시간 표시 */
      var meta = document.getElementById('sc-result-meta');
      if(meta){
        var len = origOut.value.length;
        var min = Math.round(len / 200 * 10) / 10;
        meta.textContent = len.toLocaleString() + '자 · 약 ' + min + '분';
      }
    }
  });
  observer.observe(origOut, {characterData:true, subtree:true, childList:true});
})();
